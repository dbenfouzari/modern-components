## Basic usage

<Playground>
  {() => {
    class Example extends React.Component {
      constructor (props) {
        this.state = {
          value: new Date(2019, 0, 24)
        }
        
        this.handleChange = this.handleChange.bind(this);
      }
      
      handleChange(nextDate) {
        this.setState({ value: nextDate });
      }
      
      render () {
        return (
          <React.Fragment>
            <span>Value : {this.state.value.toLocaleDateString()}</span>
            <Calendar
              value={this.state.value}
              onChange={this.handleChange}
              minDate={new Date(2019, 0, 9)}
            />
          </React.Fragment>
        )
      }
    }
    
    return <Example />
  }}
</Playground>

## Range

<Playground>
  {() => {
    class Example extends React.Component {
      constructor (props) {
        this.state = {
          value: {
            startDate: null,
            endDate: null
          }
        }
        
        this.handleChange = this.handleChange.bind(this);
      }
      
      handleChange(nextDate) {
        this.setState({ value: nextDate });
      }

      render () {
        const { value: { startDate, endDate } } = this.state;
        
        const startString = startDate ? startDate.toLocaleDateString() : '-';
        const endString = endDate ? endDate.toLocaleDateString() : '-';

        return (
          <React.Fragment>
            <div>start date : {startString}</div>
            <div>end date : {endString}</div>
            <Calendar
              value={this.state.value}
              onChange={this.handleChange}
              minDate={new Date(2018, 0, 9)}
              numberOfMonths={2}
            />
          </React.Fragment>
        )
      }
    }
    
    return <Example />
  }}
</Playground>
