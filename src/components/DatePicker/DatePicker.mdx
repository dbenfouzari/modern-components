## Basic usage

<Playground>
  {() => {
    class Example extends React.Component {
      constructor (props) {
        this.state = {
          value: new Date()
        }
        
        this.handleChange = this.handleChange.bind(this);
      }
      
      handleChange(nextDate) {
        this.setState({ value: nextDate });
      }
      
      render () {
        return <DatePicker value={this.state.value} onChange={this.handleChange} />
      }
    }
    
    return <Example />
  }}
    
</Playground>
