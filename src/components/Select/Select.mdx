## Basic usage

<Playground>
  {() => {
    class Example extends React.Component {
      constructor (props) {
        super(props);
        this.state = {
          value: []
        }
        this.handleChange = this.handleChange.bind(this);
      }

      handleChange(nextValue) {
        this.setState({ value: nextValue });
      }
      
      render () {
        return (
          <Select
            options={
              [
                { value: 1, label: "One" },
                { value: 2, label: "Two" },
                { value: 3, label: "Three" }
              ]
            }
            value={this.state.value}
            onChange={this.handleChange}
          />
        )
      }
    }
    
    return <Example />;
  }}
</Playground>
